# UX Critique of Current Implementation

## What's Working
- Clean component architecture with good separation of concerns
- Spotlight/carousel/grid variant system is flexible
- Live badge is well-executed with animation
- Card hover states are smooth and intentional

## What Needs Improvement

### 1. Hero feels generic
- "Programs" as a `Display` heading is functional but not editorial
- The body copy reads like a feature description ("Explore showcases, interviews...") rather than an invitation
- No visual tension or intrigue — it reads like a category page header on a SaaS platform

### 2. Insufficient breathing room
- `Section` with `rhythm="compact"` on the hero compresses the opening moment
- `mb-14` on section titles is decent but sections themselves need more vertical separation
- The transition from hero → spotlight feels abrupt

### 3. Spotlight could be more cinematic
- The aspect ratio `4/3 → 21/9` is good but the text overlay area (`max-w-3xl`) could be wider on large screens
- "Explore Program →" CTA feels like a button label, not an editorial invitation
- The scrim gradient could be more refined (currently a simple 3-stop)
- Gap between spotlight and next section needs more air

### 4. Card information density
- Poster cards show brand + sessions on one meta line, then heading, then speakers — this is fine but the hierarchy between meta and heading could be stronger
- The `space-y-1` on card copy is too tight for a premium feel
- `pt-4` between image and copy could be more generous

### 5. Section hierarchy is flat
- Every `<Section>` has the same padding/rhythm, making "Live Now" and "Coming Soon" feel equal weight to "Explore Programs"
- No visual differentiation between section types (all white background, same spacing)

### 6. Responsive gaps
- Mobile cards in the carousel (`Grid variant="lookbook"`) may feel cramped
- Hero text `max-w-2xl` is fine on desktop but the `mt-6` gap between heading and body is small on mobile

import { Section, Container, Grid } from "@/components/layout"
import { Headline } from "@/components/ui/Typography"

import { ProgramCard, type ProgramCardProps } from "./ProgramCard"
import { ProgramGrid } from "./ProgramGrid"

// Reuse the card's program shape — single source of truth, no new types.
type Program = ProgramCardProps["program"]

type ProgramSectionProps = {
  title?: string
  variant?: "spotlight" | "carousel" | "grid"
  programs: Program[]
}

export function ProgramSection({
  title,
  variant = "grid",
  programs,
}: ProgramSectionProps) {
  if (!programs.length) return null

  return (
    <Section>
      <Container size="wide">
        {title && (
          <Headline className="mb-10 md:mb-16">{title}</Headline>
        )}

        {variant === "spotlight" && (
          <div className="mb-8 md:mb-12">
            <ProgramCard program={programs[0]} variant="spotlight" />
          </div>
        )}

        {variant === "carousel" && (
          <Grid variant="lookbook">
            {programs.map((program) => (
              <ProgramCard key={program.id} program={program} variant="poster" />
            ))}
          </Grid>
        )}

        {variant === "grid" && <ProgramGrid programs={programs} />}
      </Container>
    </Section>
  )
}

import Image from "next/image"

import { cn } from "@/lib/utils"

// ─────────────────────────────────────────────────────────────────────────────
// Types
// ─────────────────────────────────────────────────────────────────────────────

type ProgramBrand = {
  id: number
  name: string
  slug: string
  logo_url?: string
}

type ProgramFeaturedSession = {
  id: number
  name: string
  session_type?: string
  media_type?: string
  status?: string
  thumbnail_url?: string
}

type Program = {
  id: number
  name: string
  slug: string
  description?: string

  status?: string

  cover_image_url?: string

  session_count?: number
  speaker_count?: number

  brand?: ProgramBrand

  featured_session?: ProgramFeaturedSession
}

export type ProgramCardProps = {
  program: Program
  variant?: "poster" | "spotlight"
  showBrand?: boolean
  className?: string
}

// ─────────────────────────────────────────────────────────────────────────────
// Helpers
// ─────────────────────────────────────────────────────────────────────────────

const isLive = (status?: string) => status === "live"

/** "1 session" · "8 sessions" — count is a value signal, never an apology. */
function countLabel(count: number | undefined, singular: string): string | null {
  if (count == null || count <= 0) return null
  return `${count} ${count === 1 ? singular : `${singular}s`}`
}

/** Joins editorial meta fragments with the system "·" separator. */
function metaLine(parts: Array<string | null | undefined>): string {
  return parts.filter(Boolean).join(" · ")
}

// ─────────────────────────────────────────────────────────────────────────────
// Live badge — the single sanctioned non-neutral accent (status.live token)
// ─────────────────────────────────────────────────────────────────────────────

function LiveBadge({ className }: { className?: string }) {
  return (
    <span
      className={cn(
        "inline-flex items-center gap-1.5 rounded-full",
        "bg-red-500 px-2.5 py-1",
        "text-white text-xs font-medium uppercase tracking-wide",
        className
      )}
    >
      <span className="h-1.5 w-1.5 rounded-full bg-white animate-pulse" />
      Live
    </span>
  )
}

// ─────────────────────────────────────────────────────────────────────────────
// ProgramCard
// ─────────────────────────────────────────────────────────────────────────────

export function ProgramCard({
  program,
  variant = "poster",
  showBrand = true,
  className,
}: ProgramCardProps) {
  const href = `/programs/${program.slug}`
  const cover = getImageUrl(program.cover_image_url)
  const live = isLive(program.status)

  const brandName = showBrand ? program.brand?.name : undefined
  const sessions = countLabel(program.session_count, "session")
  const speakers = countLabel(program.speaker_count, "speaker")

  // ── Spotlight — large editorial hero, text over scrim ───────────────────────
  if (variant === "spotlight") {
    const statLine = metaLine([sessions, speakers])

    return (
      <Link
        href={href}
        className={cn("group block", className)}
        aria-label={`Explore program: ${program.name}`}
      >
        <article
          className={cn(
            "relative w-full overflow-hidden rounded-3xl bg-neutral-100",
            "aspect-[3/4] sm:aspect-[4/3] md:aspect-[2/1] lg:aspect-[21/9]"
          )}
        >
          {/* Cover */}
          <Image
            src={cover}
            alt={program.name}
            fill
            priority
            sizes="(min-width: 1024px) 1280px, 100vw"
            className={cn(
              "object-cover",
              "transition-transform duration-700 ease-out",
              "group-hover:scale-[1.03]"
            )}
          />

          {/* Cinematic readability scrim */}
          <div
            className={cn(
              "absolute inset-0",
              "bg-gradient-to-t from-black/80 via-black/30 via-40% to-transparent"
            )}
          />

          {/* Live badge */}
          {live && (
            <LiveBadge className="absolute left-6 top-6 md:left-8 md:top-8" />
          )}

          {/* Text overlay */}
          <div
            className={cn(
              "absolute inset-x-0 bottom-0",
              "flex flex-col gap-4 md:gap-5",
              "p-6 md:p-10 lg:p-14",
              "max-w-4xl"
            )}
          >
            {brandName && (
              <Meta tone="inverse" className="uppercase tracking-widest">
                {brandName}
              </Meta>
            )}

            <Display
              as="h2"
              tone="inverse"
              className="transition-opacity duration-200 group-hover:opacity-80"
            >
              {program.name}
            </Display>

            {(statLine || program.featured_session?.name) && (
              <div className="flex flex-col gap-1.5">
                {statLine && (
                  <Caption variant="plain" tone="inverse">
                    {statLine}
                  </Caption>
                )}
                {program.featured_session?.name && (
                  <Caption variant="plain" tone="inverse" className="line-clamp-1 opacity-80">
                    Featuring · {program.featured_session.name}
                  </Caption>
                )}
              </div>
            )}

            {/* CTA — visual only; the whole card is the link */}
            <span
              className={cn(
                "mt-3 inline-flex items-center gap-2",
                "text-sm uppercase tracking-widest text-white/90",
                "transition-all duration-200 group-hover:text-white group-hover:gap-3"
              )}
            >
              View Program
              <span
                aria-hidden
                className="transition-transform duration-200 group-hover:translate-x-1"
              >
                →
              </span>
            </span>
          </div>
        </article>
      </Link>
    )
  }

  // ── Poster (default) — image-first discovery card ───────────────────────────
  const posterMeta = metaLine([brandName, sessions])

  return (
    <Link href={href} className={cn("group block", className)}>
      <article className="flex flex-col">
        {/* Cover */}
        <div className="relative aspect-[3/2] w-full overflow-hidden rounded-2xl bg-neutral-100">
          <Image
            src={cover}
            alt={program.name}
            fill
            sizes="(min-width: 1024px) 30vw, (min-width: 640px) 45vw, 80vw"
            className={cn(
              "object-cover",
              "transition-transform duration-700 ease-out",
              "group-hover:scale-[1.03]"
            )}
          />

          {live && <LiveBadge className="absolute right-3 top-3" />}
        </div>

        {/* Copy */}

            <Meta className="uppercase tracking-wide">{posterMeta}</Meta>

            {program.name}
          </CardHeading>

          {speakers && (
            <Caption variant="plain" tone="secondary">
              {speakers}
            </Caption>
          )}
        </div>
      </article>
    </Link>
  )
}

# UX Critique

## Current State

The poster card currently shows:

1. **Meta line**: Brand · Session count
2. **Heading**: Program name
3. **Caption**: Speaker count

This gives users a sense of *scale* (how many sessions, how many speakers) but not *substance* (what's actually inside). A user scanning the grid can see "8 sessions" but has no signal about whether those sessions are relevant to them.

## The Gap

The `featured_session.name` is the single strongest content signal available — it tells users what the editorial highlight is. It answers "what will I actually watch?" rather than just "how much is there?"

Currently this field is only surfaced in the spotlight variant, which means only one program on the entire page communicates its content. Every other card is opaque until clicked.

## What's Working

- The meta line (brand + count) is appropriately lightweight
- The heading is clearly the primary focal point
- Speaker count as tertiary info is correctly de-emphasized
- Spacing (`space-y-2 pt-5`) gives adequate breathing room

## What Needs Refinement

1. **No content preview** — the card tells you *about* the program but not *what's in it*
2. **Speaker count alone as the secondary line** feels like metadata rather than an invitation
3. **The gap between heading and speaker count** is visually flat — both are single-line elements with no intermediate hierarchy step

---

# Recommended Refinements

| Element | Current | Proposed |
|---------|---------|----------|
| Meta (top) | Brand · Sessions | Brand · Sessions (unchanged) |
| Heading | Program name | Program name (unchanged) |
| Featured session | Not shown | "Featured" label + session name, clamped to 2 lines |
| Speaker count | Directly below heading | Below featured session, remains tertiary |

### Spacing adjustments

- Increase `space-y-2` to `space-y-2.5` — the additional featured session line needs slightly more vertical rhythm to avoid feeling cramped
- The featured session block uses a tighter internal gap (`gap-0.5`) between label and name so they read as one unit
- `line-clamp-2` on the session name prevents runaway wrapping while still allowing enough context on mobile

### Typography choices

- "Featured" label: use `Meta` (same component as the top line) for consistency, but at reduced opacity to stay subordinate to the heading
- Session name: use `Caption` with default tone — visually lighter than the heading but readable as content

# Why This Hierarchy Improves Discoverability

1. **Content signal before click** — Users scanning a grid of 6–12 cards can now read the featured session name and make a relevance judgment without opening each program. This reduces cognitive load and increases click confidence.

2. **Progressive disclosure within the card** — The hierarchy flows from *who* (brand) → *what* (program name) → *what's inside* (featured session) → *scale* (speakers). Each level answers a progressively more specific question.

3. **The "Featured" label creates a content promise** — It signals editorial curation. Users understand that someone chose this session as the highlight, which implies quality and intentionality — the same pattern Netflix uses with "Top 10" or Masterclass uses with session previews.

4. **Speaker count drops to true tertiary** — With the featured session providing substance, speaker count becomes a soft credibility signal rather than the only secondary element. The `opacity-70` treatment correctly communicates its reduced importance without removing it.

5. **Mobile resilience** — `line-clamp-2` ensures the featured session name never dominates the card on small screens. The `gap-0.5` between "Featured" and the session name keeps them visually bound as one unit, so even when the name wraps, the card doesn't feel like it has too many disconnected lines.

const API_BASE_URL = process.env.NEXT_PUBLIC_API_URL || "http://localhost:3002"

export async function getPrograms(params?: Record<string, string>) {
  const query = new URLSearchParams(params).toString()

  const res = await fetch(
    `${API_BASE_URL}/api/v2/storefront/programs?${query}`,
    { cache: "no-store" }
  )

  return res.json()
}

export async function getProgram(slug: string) {
  const res = await fetch(
    `${API_BASE_URL}/api/v2/storefront/programs/${slug}`,
    { cache: "no-store" }
  )

  if (!res.ok) {
    throw new Error(`Failed to fetch program: ${slug}`)
  }

  return res.json()
}

# Explanation of Decisions

### API Client (`lib/api/programs.ts`)

- Follows the same pattern as `getProducts` — direct `fetch` with the base URL from environment variables
- Uses `cache: "no-store"` consistent with the existing `getPrograms` function
- Throws on non-ok responses so the page can catch and call `notFound()`
- No new abstractions, no wrapper utilities

### Page Structure

1. **Hero** — Reuses the same cinematic pattern from the spotlight card (rounded-3xl, scrim, text overlay) but adapted for a detail page context. The CTA links directly to the featured session (or first session as fallback).

2. **Sessions** — Adaptive grid based on count. No timeline, no schedule, no agenda. Each card is a simple image-first discovery unit linking to the session detail page.

3. **Speakers** — Lightweight avatar grid. Centered layout with minimal info. No bios, no links (can be added later).

4. **About** — Two-column on desktop (prose + metadata), stacked on mobile. Provides the full description and key stats for users who scroll past the hero.

5. **More From Brand** — Placeholder with a neutral background and "coming soon" message. Ready for API integration later.

### Conventions Followed

- `Section` + `Container size="wide"` for every section
- Typography components (`Display`, `Headline`, `CardHeading`, `Caption`, `Meta`, `Body`) from the design system
- `getImageUrl` helper for all image URLs
- `cn()` for conditional class merging
- `rounded-2xl` / `rounded-3xl` for card/hero radii
- `group` + `group-hover:` for card interactions
- `line-clamp-*` for text truncation
- Same spacing patterns (`mb-10 md:mb-16` for section titles, `space-y-2 pt-5` for card copy)

const API_BASE_URL = process.env.NEXT_PUBLIC_API_URL || "http://localhost:3002"

export async function getPrograms(params?: Record<string, string>) {
  const query = new URLSearchParams(params).toString()

  const res = await fetch(
    `${API_BASE_URL}/api/v2/storefront/programs?${query}`,
    { cache: "no-store" }
  )

  return res.json()
}

export async function getProgram(slug: string) {
  const res = await fetch(
    `${API_BASE_URL}/api/v2/storefront/programs/${slug}`,
    { cache: "no-store" }
  )

  if (!res.ok) {
    throw new Error(`Failed to fetch program: ${slug}`)
  }

  return res.json()
}
